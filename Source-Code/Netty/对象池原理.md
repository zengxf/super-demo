## 单元测试
```java
public class ObjPoolTest {
    private static final Recycler<User> RECYCLER = new Recycler<User>() {
        @Override
        protected User newObject(Handle<User> handle) {
            User user = new User(handle);
            user.name = "init-88";
            return user;
        }
    };

    static final class User {
        public String name;
        private final Recycler.Handle<User> handle;

        public User(Recycler.Handle<User> handle) {
            this.handle = handle;
        }

        public void recycle() {
            handle.recycle(this);
        }
    }

    @Test
    public void test() {
        User user1 = RECYCLER.get();    // 1、从对象池获取 User 对象
        user1.name = "hello";           // 2、设置 User 对象的属性
        user1.recycle();                // 3、回收对象到对象池
        // user1.recycle();             // 如果不调用 recycle(), 则会创建新的对象

        User user2 = RECYCLER.get();    // 4、从对象池获取对象
        System.out.println(user2.name);
        System.out.println(user1 == user2);
    }
}
```
